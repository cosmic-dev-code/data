### ========================================= ###
###### ===--- Instalacion por (CDN) ---=== ######
### ========================================= ###

<!-- Colocamos el CDN. -->

```html
    <script src="https://cdn.jsdelivr.net/npm/axios/dist/axios.min.js"></script>
```

### ========================================= ###
###### ===--- Instalacion por (npm) ---=== ######
### ========================================= ###

<!-- Instalamos por (npm). -->

```bat
	npm install axios
```

<!-- Ahora lo importamos en nuestro archivo JavaScript. -->

```js
	const axios = require('axios');
```

<!-- O de esta forma. -->

```js
	import axios from 'axios';
```
